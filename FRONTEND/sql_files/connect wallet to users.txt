-- Drop existing triggers first (if you want to replace them)
DROP TRIGGER IF EXISTS sync_wallet_to_user;
DROP TRIGGER IF EXISTS sync_wallet_insert;

-- First trigger: AFTER UPDATE
DELIMITER $$
CREATE TRIGGER sync_wallet_to_user
AFTER UPDATE ON wallets
FOR EACH ROW
BEGIN
    UPDATE users 
    SET balance = NEW.available_balance,
        spent = NEW.spent_balance,
        updated_at = NOW()
    WHERE id = NEW.user_id;
END$$
DELIMITER ;

-- Second trigger: AFTER INSERT
DELIMITER $$
CREATE TRIGGER sync_wallet_insert
AFTER INSERT ON wallets
FOR EACH ROW
BEGIN
    UPDATE users 
    SET balance = NEW.available_balance,
        spent = NEW.spent_balance,
        updated_at = NOW()
    WHERE id = NEW.user_id;
END$$
DELIMITER ;
